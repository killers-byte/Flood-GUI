    local function parseExpiry(expiry)
        if expiry == "permanent" then return 9999999999 end
        if type(expiry) == "string" then
            local year, month, day = string.match(expiry, "^(%d%d%d%d)-(%d%d)-(%d%d)$")
            if year and month and day then
                year, month, day = tonumber(year), tonumber(month), tonumber(day)
                local isLeap = function(y) return (y % 4 == 0 and y % 100 ~= 0) or (y % 400 == 0) end
                local daysInMonth = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}
                local days = 0
                for y = 1970, year - 1 do days = days + (isLeap(y) and 366 or 365) end
                for m = 1, month - 1 do days = days + daysInMonth[m]; if m == 2 and isLeap(year) then days = days + 1 end end
                days = days + (day - 1)
                return (days * 86400) + 86399 - (7 * 3600)
            end
        end
        return nil
    end

-- ?? PASSIVE REPORT DETECTION (No. 7)
local playerPositionCache = {}
local function checkPassiveReport()
    if not CONFIG.PASSIVE_REPORT_DETECT then return end
    local now = os.clock()
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= Player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local root = p.Character.HumanoidRootPart
            -- Hanya deteksi yang di lobby (X < 50) dan bukan di lift (Z <= 70)
            if root.Position.X < 50 and root.Position.Z <= 70 then
                local userId = p.UserId
                if not playerPositionCache[userId] then
                    playerPositionCache[userId] = {pos = root.Position, time = now}
                else
                    local dist = (root.Position - playerPositionCache[userId].pos).Magnitude
                    if dist < 0.5 and (now - playerPositionCache[userId].time) > 5 then
                        STATE.SuspiciousSpectator = true
                        notify("?? Player " .. p.Name .. " diam di lobby >5 detik (mungkin merekam)!", "PASSIVE REPORT")
                        -- Bisa langsung ghost atau disconnect jika diatur
                        if CONFIG.NUCLEAR_ESCAPE then
                            selfDestruct()
                        elseif not STATE.isGhostMode then
                            activateGhostMode()
                        end
                    else
                        playerPositionCache[userId] = {pos = root.Position, time = now}
                    end
                end
            end
        end
    end
end

-- ?? ULTIMATE ADMIN DETECTION (FE2 GROUP 3196771)
local adminDatabaseModule = nil
local function getAdminDatabase()
    if adminDatabaseModule then return adminDatabaseModule end
    for _, obj in ipairs(game:GetDescendants()) do
        if obj:IsA("ModuleScript") then
            local name = obj.Name:lower()
            if name:find("admin") or name:find("staff") or name:find("mod") or name:find("permission") then
                local success, module = pcall(require, obj)
                if success and type(module) == "table" then
                    if module.IsAdmin or module.Admins or module.Staff or module.Moderators then
                        adminDatabaseModule = module
                        return module
                    end
                end
            end
        end
    end
    return nil
end

local function isAdmin(p)
    if not p then return false end
    if p.UserId == game.CreatorId then return true end
    local okRoblox, isRobloxAdmin = pcall(function() return p:IsInGroup(1200769) end)
    if okRoblox and isRobloxAdmin then return true end

    local okFe2, role = pcall(function() return p:GetRoleInGroup(3196771) end)
    if okFe2 and role then
        local lowerRole = role:lower()
        if lowerRole:find("admin") or lowerRole:find("mod") or lowerRole:find("owner") or lowerRole:find("staff") or lowerRole:find("developer") or lowerRole:find("crazyblox") or lowerRole:find("team") then return true end
        local okRank, rankNumber = pcall(function() return p:GetRankInGroup(3196771) end)
        if okRank and type(rankNumber) == "number" and rankNumber >= 100 then return true end
    end

    local db = getAdminDatabase()
    if db then
        if db.IsAdmin and type(db.IsAdmin) == "function" then
            local status, result = pcall(db.IsAdmin, p)
            if status and result then return true end
        end
        if db.Admins and type(db.Admins) == "table" then
            if db.Admins[p.UserId] or db.Admins[tostring(p.UserId)] then return true end
        end
        if db.Staff and type(db.Staff) == "table" then
            if db.Staff[p.UserId] or db.Staff[tostring(p.UserId)] then return true end
        end
        if db.Moderators and type(db.Moderators) == "table" then
            if db.Moderators[p.UserId] or db.Moderators[tostring(p.UserId)] then return true end
        end
    end

    local pGui = p:FindFirstChild("PlayerGui")
    if pGui then
        if pGui:FindFirstChild("HDAdminGUIs") or pGui:FindFirstChild("Adonis_UI") or pGui:FindFirstChild("KohlsAdmin") or pGui:FindFirstChild("AdminUI") then return true end
    end

    local leaderstats = p:FindFirstChild("leaderstats")
    if leaderstats then
        for _, stat in ipairs(leaderstats:GetChildren()) do
            local lowerName = stat.Name:lower()
            if lowerName == "admin" or lowerName == "mod" or lowerName == "rank" or lowerName == "role" or lowerName == "staff" then
                local val = tostring(stat.Value):lower()
                if val == "admin" or val == "mod" or val == "owner" or val == "dev" or val == "staff" or val == "creator" or val == "headadmin" then return true end
            end
        end
    end

    elseif keyExpireTime > GetRealTime() + 315360000 then dKeyLabel.Text = "?? Key: PERMANENT"; dKeyLabel.TextColor3 = M_ACCENT

        dKeyLabel.Text = "?? Expires In: "..(days>0 and days.."d " or "")..hours.."h "..minutes.."m"; dKeyLabel.TextColor3 = Color3.fromRGB(255,200,100) end
    dMapLabel.Text = "??? Map: "..(Stats.currentMap~="" and Stats.currentMap or "Waiting...")
    dTimeLabel.Text = "?? Time: "..math.floor((os.clock()-Stats.sessionStart)/60).."m"
    local hrs = (os.clock()-Stats.sessionStart)/3600
    dRateLabel.Text = "? Rate: "..((hrs>0 and Stats.mapsCompleted>0) and math.floor(Stats.mapsCompleted/hrs) or 0).." maps/hr"
    if STATE.panicActive then dStatusLabel.Text = "?? Status: PANIC"; dStatusLabel.TextColor3 = Color3.fromRGB(255,80,80)
    elseif STATE.TAS_RUNNING then dStatusLabel.Text = "?? Status: TAS PLAYING"; dStatusLabel.TextColor3 = M_ACCENT
    elseif STATE.AUTO_QUEUE_ENABLED then dStatusLabel.Text = "?? Status: Auto Queue"; dStatusLabel.TextColor3 = Color3.fromRGB(100,200,255)
    else dStatusLabel.Text = "?? Status: Idle"; dStatusLabel.TextColor3 = M_TEXT_D end
    if STATE.isGhostMode then dGhostLabel.Text = "?? Ghost Mode: ACTIVE"; dGhostLabel.TextColor3 = Color3.fromRGB(255,50,50)
    else dGhostLabel.Text = "?? Ghost Mode: Inactive"; dGhostLabel.TextColor3 = Color3.fromRGB(180,180,180) end
    local admins = getAdminPlayers()
    if #admins > 0 then local txt = {}; for _, adm in ipairs(admins) do table.insert(txt, adm.Name) end
        dAdminLabel.Text = "??? Admins: "..table.concat(txt,", "); dAdminLabel.TextColor3 = Color3.fromRGB(255,80,80)
    else dAdminLabel.Text = "??? Admins: None"; dAdminLabel.TextColor3 = Color3.fromRGB(100,255,100) end
    -- Suspicious indicator
    if STATE.SuspiciousSpectator then
        dSuspiciousLabel.Text = "?? Suspicious: YES"; dSuspiciousLabel.TextColor3 = Color3.fromRGB(255,200,0)
    else
        dSuspiciousLabel.Text = "?? Suspicious: No"; dSuspiciousLabel.TextColor3 = M_TEXT_D
    end
end
